    ```javascript
    // compositions.js — EAN Composition CRUD module
    // Allows defining which components make up a composite product.

    const compositions = {
        currentParentEan: null,
        currentRows: [],  // [{component_ean, quantity}] pending save

        // ── Search/Load ─────────────────────────────────────────

        async searchProducts(query) {
            const products = await api.request(`/api/products?q=${encodeURIComponent(query)}`);
            return products;
        },

        async loadForParent(parentEan) {
            try {
                const rows = await api.request(`/api/compositions/${encodeURIComponent(parentEan)}`);
                this.currentParentEan = parentEan;
                this.currentRows = rows.map(r => ({ id: r.id, component_ean: r.component_ean, component_name: r.component_name, quantity: r.quantity }));
                this.renderRows();
            } catch (e) {
                toast.show('Failed to load composition: ' + e.message, 'error');
            }
        },

        // ── Save ────────────────────────────────────────────────

        async save() {
            if (!this.currentParentEan) {
                toast.show('Select a parent product first', 'error');
                return;
            }
            const payload = this.currentRows.map(r => ({ component_ean: r.component_ean, quantity: r.quantity }));
            try {
                const result = await api.request(`/api/compositions/${encodeURIComponent(this.currentParentEan)}`, {
                    method: 'PUT',
                    body: JSON.stringify(payload),
                });
                toast.show(`Saved ${result.component_count} component(s)`, 'success');
                await this.loadForParent(this.currentParentEan);
            } catch (e) {
                // api.request throws on 4xx — message from FastAPI detail field
                toast.show(e.message || 'Save failed', 'error');
            }
        },

        // ── Add/Remove rows ─────────────────────────────────────

        addRow() {
            this.currentRows.push({ component_ean: '', quantity: 1 });
            this.renderRows();
            // Focus the last EAN input
            const inputs = document.querySelectorAll('#comp-rows .comp-ean-input');
            if (inputs.length > 0) inputs[inputs.length - 1].focus();
        },

        removeRow(index) {
            this.currentRows.splice(index, 1);
            this.renderRows();
        },

        // ── Render ──────────────────────────────────────────────

        renderRows() {
            const container = document.getElementById('comp-rows');
            if (!container) return;
            if (this.currentRows.length === 0) {
                container.innerHTML = '<tr><td colspan="3" class="comp-empty">No components defined. Click "Add Row" to start.</td></tr>';
                return;
            }
            container.innerHTML = this.currentRows.map((row, i) => `
                <tr>
                    <td><input class="comp-ean-input input-field" type="text" value="${sanitize(row.component_ean)}" data-index="${i}" placeholder="Component EAN" /></td>
                    <td><input class="comp-qty-input input-field" type="number" value="${sanitize(row.quantity)}" data-index="${i}" min="1" style="width:80px"/></td>
                    <td><span class="comp-name-label">${sanitize(row.component_name || '')}</span></td>
                    <td><button class="btn-icon comp-remove-btn" data-index="${i}" title="Remove">✕</button></td>
                </tr>
            `).join('');

            // Attach input change listeners
            container.querySelectorAll('.comp-ean-input').forEach(input => {
                input.addEventListener('change', (e) => {
                    const idx = parseInt(e.target.dataset.index);
                    this.currentRows[idx].component_ean = e.target.value.trim();
                    this.currentRows[idx].component_name = '';  // cleared until save resolves
                });
            });
            container.querySelectorAll('.comp-qty-input').forEach(input => {
                input.addEventListener('change', (e) => {
                    const idx = parseInt(e.target.dataset.index);
                    this.currentRows[idx].quantity = Math.max(1, parseInt(e.target.value) || 1);
                });
            });
            container.querySelectorAll('.comp-remove-btn').forEach(btn => {
                btn.addEventListener('click', (e) => {
                    const idx = parseInt(e.target.dataset.index);
                    this.removeRow(idx);
                });
            });
        },

        renderParentLabel(parentEan) {
            const el = document.getElementById('comp-parent-label');
            if (el) el.textContent = parentEan ? `Parent: ${parentEan}` : 'No parent selected';
        },

        // ── Parent search ────────────────────────────────────────

        async handleParentSearch(e) {
            const query = e.target.value.trim();
            if (query.length < 2) {
                document.getElementById('comp-search-results').innerHTML = '';
                return;
            }
            try {
                const products = await this.searchProducts(query);
                const results = document.getElementById('comp-search-results');
                if (!results) return;
                if (products.length === 0) {
                    results.innerHTML = '<li class="comp-no-results">No products found</li>';
                    return;
                }
                results.innerHTML = products.slice(0, 10).map(p => `
                    <li class="comp-result-item" data-ean="${sanitize(p.ean)}">
                        <span class="comp-result-ean">${sanitize(p.ean)}</span>
                        <span class="comp-result-name">${sanitize(p.name)}</span>
                    </li>
                `).join('');
                results.querySelectorAll('.comp-result-item').forEach(li => {
                    li.addEventListener('click', async () => {
                        const ean = li.dataset.ean;
                        document.getElementById('comp-parent-search').value = ean;
                        document.getElementById('comp-search-results').innerHTML = '';
                        await this.loadForParent(ean);
                        this.renderParentLabel(ean);
                    });
                });
            } catch (e) {
                toast.show('Search failed: ' + e.message, 'error');
            }
        },

        // ── Init ────────────────────────────────────────────────

        init() {
            // Called by app-init.js when compositions view is navigated to
            const searchInput = document.getElementById('comp-parent-search');
            if (searchInput) {
                searchInput.addEventListener('input', (e) => this.handleParentSearch(e));
            }
            const addBtn = document.getElementById('comp-add-row-btn');
            if (addBtn) {
                addBtn.addEventListener('click', () => this.addRow());
            }
            const saveBtn = document.getElementById('comp-save-btn');
            if (saveBtn) {
                saveBtn.addEventListener('click', () => this.save());
            }
            this.renderRows();
        }
    };

    window.compositions = compositions;
    ```
